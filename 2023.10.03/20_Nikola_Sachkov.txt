
SELECT - Извлича данни от базата
UPDATE - Променя данни в базата
DELETE - Изтрива данни от базата
INSERT INTO - Вмъква нови данни в база данни
CREATE DATABASE - Създава нова база данни
ALTER DATABASE - Променя структурата на база данни
CREATE TABLE - Създава нова таблица в база данни
ALTER TABLE - Променя структурата на таблица в база данни (добавя колона например)
DROP TABLE - Изтрива таблица от база данни
