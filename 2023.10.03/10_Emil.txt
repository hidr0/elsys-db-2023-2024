SELECT - Избира (извлича) информация (данни) от таблица (Пр. Оценки)
UPDATE - Променя информация (данни) от дадена таблица (Пр. Оценки)
DELETE - Изтрива информация (данни) от дадена таблица (Пр. Изтрива оценки или ученици)
INSERT INTO - Добавя информация (данни) в нова таблица (Пр. Добавя оценки или ученици)
CREATE DATABASE - Създава нова база данни (Пр. Дневник)
ALTER DATABASE - Позволява модифициране на дадена база (Пр. Промяна на дневника)
CREATE TABLE - Съзадава нова таблица (Пр. Учители, които въвеждат оценки)
ALTER TABLE - Позволява модифициране на дадена таблица (Пр. Промяна на учениците)
DROP TABLE - Изтрива таблица (Пр. Ученици)
