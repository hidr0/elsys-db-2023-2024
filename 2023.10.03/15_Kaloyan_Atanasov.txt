SELECT - взима дата от таблиците в датабазата

UPDATE - update-ва (заменя) стойностите на дадена дата

DELETE - изтрива дата от дадена таблица в датабазата

INSERT INTO - добавя нови редове в таблицата

CREATE DATABASE - създаваме датабаза

ALTER DATABASE - модифицираме датабазата = можем да сменяме името ѝ, да задаваме опции на датабазата или да модифицираме (променяме) файловете ѝ

CREATE TABLE - създаваме таблица

ALTER TABLE - добавя, изтрива или модифицира колоните във вече съществуваща таблица

DROP TABLE - изтриваме таблица
