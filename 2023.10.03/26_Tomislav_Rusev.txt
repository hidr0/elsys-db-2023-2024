SELECT - Взема някаква data в базата данни(Пример: y1)
UPDATE - Прави промяна по data-та в базата данни(Пример: y1 става y30)
DELETE - Изтрива някаква data от базата данни(Пример: y30)
INSERT INTO - Добавя нова data в базата данни(Пример: Добавя като оценка y31)
CREATE DATABASE - Създава нова база данни
ALTER DATABASE - Модифицира таблица или свойства на съществуваща база данни(Пример: променя името на базата данни)
CREATE TABLE - Създава нова таблица
ALTER TABLE - Прави промени по таблица(като например да промени редовете)
DROP TABLE - Изтрива таблица
