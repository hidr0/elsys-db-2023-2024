SELECT - Тази команда се използва за извличане на информация от база данни

UPDATE - Служи за промяна на съществуващи стойности в дадена таблица

DELETE - Служи за изтриване на съществуващи стойности в дадена таблица

INSERT INTO - Тази команда се използва за добавяне на стойности в дадена таблица

CREATE DATABASE - Използва се за създаване на нова SQL база данни

ALTER DATABASE - Използва се за модифициране или възстановяване на съществуваща база данни

CREATE TABLE - Използва се за създаване на нова таблица в дадена база данни

ALTER TABLE - Използва се за модифициране на колони на съществуваща таблица

DROP TABLE - Използва се за изтриване на таблица от дадена база данни
