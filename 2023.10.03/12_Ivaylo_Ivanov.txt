SELECT - селектира данни от таблица в базата данни, за да се употребят

UPDATE - актуализира/променя налични данни в таблица от базата данни

DELETE - изтрива/заличава данни от таблица в базата данни

INSERT INTO - вкарване на нова информация в таблицата от базата даннни

CREATE DATABASE - създава нова база данни

ALTER DATABASE - прави промени по вече съществуваща база данни

CREATE TABLE - създава нова таблица в някоя си база от данни

ALTER TABLE - променя/модифицира информацията вече създадена таблица

DROP TABLE - sends the table into oblivion/така де, изтирва таблицата
