SELECT - Select-ва нянаква информация от база данни.
UPDATE - Update-ва някаква информация от база данни.
DELETE - Delete-ва някаква информация от база данни. 
INSERT INTO - Добавя нова информация в база данни.
CREATE DATABASE - Създава нова база данни.
ALTER DATABASE - Прави промени по база данни.
CREATE TABLE - Създава таблица в база данни.
ALTER TABLE - Прави промени по таблица от база данни.
DROP TABLE - Delete-ва таблица от база данни.
