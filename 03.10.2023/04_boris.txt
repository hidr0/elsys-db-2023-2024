SELECT - взима данни от база данни
UPDATE - променя данни от база данни
DELETE - изтрива данни от база данни
INSERT INTO - добавя нови данни в база данни
CREATE DATABASE - създава нова база данни
ALTER DATABASE - променя база данни
CREATE TABLE - създава нова таблица в база данни
ALTER TABLE - променя таблица от база данни
DROP TABLE - изтрива таблица от база данни
