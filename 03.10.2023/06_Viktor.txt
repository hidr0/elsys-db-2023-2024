SELECT - избира/извлича данни от дадена таблица
UPDATE - променя данни в дадена таблица
DELETE - изтрива данни от дадена таблица
INSERT INTO - добавя нови данни в дадена таблица
CREATE DATABASE - създава нова база
ALTER DATABASE - позволява манипулация на дадена база
CREATE TABLE - създава нова таблица
ALTER TABLE - позволява манипулация на дадена таблица
DROP TABLE - изтрива дадена таблица
