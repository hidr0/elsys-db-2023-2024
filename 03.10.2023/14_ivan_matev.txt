SELECT - Избира дата от база данни (В нашия случай У1, У2, и тн. от таблицата Ученици/Оценки)
UPDATE - Променя дата в база данни (В нашия случай У1, У2, и тн. от таблицата Ученици/Оценки)
DELETE - Трие дата в база данни (В нашия случай О1, О2, и тн. от таблицата Ученици/Оценки)
INSERT INTO - Добавя дата в база данни (Пример О4 за У7 от таблицата Ученици/Оценки)
CREATE DATABASE - Създава дата база(В нашия случай Дневник)
ALTER DATABASE - Променя нещо по датата база
CREATE TABLE - Създава нова таблица (Примерно Ученици/Оценки)
ALTER TABLE - Променя нещо по таблицата (Примерно слага лимит на оценките от 0-6)
DROP TABLE - Трие таблицата
