Стоян Стоянов 

SELECT - С него заявляваме с кои данни от базата ще работим , също пази резултатите.

UPDATE - Променя същестуващ запис в таблица.

DELETE - Използва се за изтриваме на (редове)данни от таблица. Изтрива рекорди.

INSERT INTO - Добавя нови редове в таблица

CREATE DATABASE - Създава  нова база данни 

ALTER DATABASE - Позволява редактирането на база данни

CREATE TABLE - Създава таблица в базата данни 

ALTER TABLE - Позволява редактирането на таблица в базата от данни

DROP TABLE -  Премахва същестуваща база от данни
