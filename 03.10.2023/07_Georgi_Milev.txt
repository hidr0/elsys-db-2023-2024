SELECT - избира данни от дадената таблица
UPDATE - променя данниите в таблицата
DELETE - изтрива от таблицата
INSERT INTO - добавя нови данни таблицата
CREATE DATABASE - създава нова база данни
ALTER DATABASE - позволява да се манипулацира базата
CREATE TABLE - създава нова таблица
ALTER TABLE - позволява да се манипулацира таблицатю
DROP TABLE - изтрива таблица
