SELECT - Избира данн от таблицата

UPDATE - променя данните в таблицата 

DELETE - изтрива данни от таблицата

INSERT INTO - добавя данни някъде в таблицата

CREATE DATABASE - прави нова база

ALTER DATABASE - дава възможност да се променя базата

CREATE TABLE - пави нова таблица 

ALTER TABLE - дава възможност да се променя таблицата 

DROP TABLE - изтрива таблицата
